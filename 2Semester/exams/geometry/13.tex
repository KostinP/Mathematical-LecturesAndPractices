    \begin{Definition}
        \[(X, \Omega) \text{ - топ пр-во}\]
        \[X \text{ - несвязное, если } \exists \text{ откр. } U_1, U_2 \subset X\]
        \[X = U_1 \cup U_2 \qq \varnothing = U_1 \cap U_2\]
        \[U_1, U_2 \neq \varnothing\]
    \end{Definition}
