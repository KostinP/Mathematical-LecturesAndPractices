  \section{Связность произведения пространств}

  \begin{theorem}
      X, Y - топ. пр-ва\\
      $X \times Y$ - связн. $\rla$ X, Y - связн.
  \end{theorem}

  \begin{remark}
      Любое конечное произведение связных топ. пр-в связно
  \end{remark}

  \begin{theorem}
      \[\prod_{i \in I} X_i \text{ - связно} \rla \forall i \in I \q X_i \text{ - связно}\]
  \end{theorem}
