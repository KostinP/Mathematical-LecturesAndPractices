  \section{Гомеоморфизм.}

  \begin{definition}
      $f: X \rightarrow Y$ - гомеоморфизм, если\\
      \begin{enumerate}
          \item f - непр.
          \item f - биекция
          \item $f^{-1}$ - непр.
      \end{enumerate}
  \end{definition}

  \begin{hypothesis}
      $\simeq$ - отношение эквив.
  \end{hypothesis}

  \begin{theorem}
      Если $(X, \Omega_X) \simeq (Y, \Omega_Y)$, то\\
      $f_*: \Omega_X \rightarrow \Omega_Y$ - биекция\\
      $f_*(U) = f(U)$
  \end{theorem}
