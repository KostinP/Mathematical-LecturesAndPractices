  \section{Вторая аксиома счётности и сепарабельность.}

  \begin{definition}
      X - обл. II А.С., если в X $\exists$ счетная база
  \end{definition}

  \begin{definition}
      X - назыв сепараб., если $\exists$ A $\subset$ X\\
      |A| $\leq \aleph_0$ и $Cl A = X$
  \end{definition}

  \begin{definition}
      A - всюду плотно, если ClA = X
  \end{definition}

  \begin{theorem}
      X - II А.С. $\ra$ X - сепараб.
  \end{theorem}
