	\begin{Utv}
			\[H, G \text{ - группы, } \q g \in G\]
			\[\varphi : G \to H \text{ - изоморфизм}\]
			\[\Ra \text{ ord } g = \text{ ord } \varphi(g)\]
			\[\text{ord } g = n \q\q g^n = e\]
			\[\varphi(g)^n = \varphi(g^n) = \varphi(e) = e \q\q \varphi(e)^2 = \varphi(e^2) = \varphi(e)\]
			\[\varphi(g)^n \os{?}{\Ra} e \Ra m \geq n\]
			\[m \in \N\]
			\[\varphi(g^m) = \varphi(g)^m = e = \varphi(e) \q \Ra g^m = e \Ra m \geq n\]
	\end{Utv}

	\begin{Definition}
			\[H < G\]
			\[H \text{ - нормальная подгруппа, если } \forall h \in H, g \in G\]
			\[g^{-1}hg \in H \text{ - сопряжение элемента h с помощью элемента g}\]
			рисунок 1
			\[H \vartriangleleft G\]
	\end{Definition}

	\begin{Utv}
			\[H \vartriangleleft G \rla \text{ - разбиение на л. и п. классы смежности по H совпадают}\]
			\[\forall g \q gH = Hg\]
	\end{Utv}

	\begin{Proof}
			\[\Ra h \in H \q\q gh \in gH\]
			\[gh = \underbrace{(g^{-1})^{-1} h g^{-1}}_{\in H} g = h_1 g\]
			\[\La g \in G , h \in H\]
			\[g^{-1}hg = h_1\]
			\[hg \in Hg = gH \Ra gh_1, h_1 \in H\]
	\end{Proof}

	\begin{Example}
		\[[g, h] = g h g^{-1} h^{-1} \text{ - коммутатор} \]
		\[g, h \in G\]
		\[K(G) = \{[g_1, h_1], ... ,[g_n, h_n], g_i, h_i \in G\} \text{ - коммутант}\] 
	\end{Example}

	\begin{proof}
			Коммутант - подгруппа
			\[K(G) < G\]
			\[[e, e] = e\]
			\[[g_1, h_1] ... [g_n, h_n]\]
			\[[g, h]^{-1} = (ghg^{-1}h^{-1})^{-1} = hgh^{-1}g^{-1} = [h, g]\]
			\[([g_1, h_1] ... [g_n, h_n])^{-1} = [h_1, g_1] ... [g_n, h_n]\]
			\[g^{-1}[g_1, h_1] ... [g_n, h_n]g = \]
			\[= (g^{-1} [g_1, h_1]g)(g^{-1}[g_2, h_2]g) ... (g^{-1}[g_n, h_n]g) \]
			\[g^{-1} g_1 h_1 g_1^{-1} h_1^{-1} g = \]
			\[ = (g^{-1} g_1 h_1 g_1^{-1} (g h_1^{-1})h_1 g^{-1})h_1^{-1}g   \]
			\[[g^{-1}g_1, h_1] \q\q [h_1, g^{-1} ]\]
	\end{proof}

	\begin{Utv}

	\begin{Proof}
		\[g_1, g_2 \in G \q\q g_1 K(G) g_2 K(G) \os{?}{=} g_2 K(G) g_1 K(G)\]
		\[g_1 g_2 K (G) = g_1 g_2 K(G) \q\q g_2 K(G) g_1 K(G) = g_2 g_1 K(G)\]
		\[[g_1, g_2] = g_1 g_2 (g_2 g_1)^{-1} \in K(G) \]
	\end{Proof}
